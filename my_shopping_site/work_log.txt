・sequelizeに関するその他のこと（オプションなど）
    -freezeTableName: true,  //テーブル名をモデルと同じにする
    -timestamps: false,  //タイムスタンプカラム(updatedAt, createdAt)を使用しない

            code: {
                allowNull: false,
                autoIncrement: true,
                primaryKey: true,
                type: Sequelize.INTEGER
